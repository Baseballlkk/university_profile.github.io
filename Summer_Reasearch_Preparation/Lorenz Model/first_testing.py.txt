# import library
import numpy as np
from matplotlib import pyplot as plt
from mpl_toolkits.mplot3d import Axes3D

# build up array
x = np.zeros(1000000)
y = np.zeros(1000000)
z = np.zeros(1000000)

# set initial condition
x[0] = pow(10,-100)
y[0] = pow(10,-100)
z[0] = pow(10,-100)
dt = pow(10,-10)

# calculate the model
for i in range(len(x)-1):
	x[i+1] = 10*(y[i]-x[i])*dt+x[i]
	y[i+1] = (28-x[i]*z[i]-y[i])*dt+y[i]
	z[i+1] = (x[i]*y[i]-(8/3)*z[i])*dt+z[i]

print(x)
print(y)
print(z)

# plot the scatter
fig=plt.figure()
ax2 = Axes3D(fig)

ax2.scatter(x,y,z)
plt.show()
